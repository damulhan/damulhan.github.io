# Flux control

(if true
  :truthy
  :falsey)

(if (= (+ 1 1) 2)
  "Then true"
  "Opitional else value")

(let [x 7]
  (cond
    (< x 2) "x is less than 2"
    (< x 10) "x is less than 10"))

(let [x 14]
  (cond
    (< x 2)  "x is less than 2"
    (< x 10) "x is less than 10"
    :else  "x is greater than or equal to 14"))

(when true
  (println "Success!")
  "Awesome too!")

(defn foo
  [c]
    (case c
      5 "c is 5"
      10 "c is 10"))
;; user=> (foo 5)

(defn foo
  [c]
    (case c
      5 "c is 5"
      10 "c is 10"
      "c isn't 5 or 10"))
;; user=> (foo 14)

# Iteration for Side Effects

(dotimes [i 7]
         (println i))

(doseq [n (range 3)]
         (println n))

(for [letter [:a :b]
             number (range 3)]
         [letter number])

(loop [i 0]
  (if (< i 10)
    (recur (inc i))
    i))

# Exceptions

(try
  (/ 2 1)
  (catch ArithmeticException e
    "divide by zero")
  (finally
    (println "cleanup")))

(try
  (throw (ex-info "There was a problem" {:detail 42}))
  (catch Exception e
    (prn (:detail (ex-data e)))))
